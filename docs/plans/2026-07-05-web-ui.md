# Web UI 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为 Docker-MCPilotS 增加单管理员 Web UI，提供仪表盘、容器管理、用户权限管理、审计日志四大模块，支持配置热加载。

**Architecture:** Web UI 作为 FastMCP 的额外 HTTP 路由同进程部署，复用现有 `DockerClient` / `SystemDiag` / `AuthConfig`。引入 `AppState` 可变引用实现热加载，新增 `AuditLogger` 记录人类与 AI Agent 的所有操作。前端采用 HTML + 原生 JS + HTMX，避免 Node 构建链。

**Tech Stack:** Python 3.11 / FastMCP custom_route / Jinja2 / itsdangerous / bcrypt / 原生 CSS（暗色科技风，CSS 变量驱动）

# core/app_state.py
from __future__ import annotations
from dataclasses import dataclass
from core.config import Settings, AuthConfig
from core.auth import PermissionChecker

@dataclass
class AppState:
    settings: Settings
    auth_config: AuthConfig
    audit_logger: "AuditLogger | None" = None

    def __post_init__(self):
        self.permission_checker = PermissionChecker(self.auth_config)

    def reload_auth(self, auth_yaml_path: str) -> None:
        new_config = AuthConfig.from_yaml(auth_yaml_path)
        self.auth_config = new_config
        self.permission_checker = PermissionChecker(new_config)

    def reload_settings(self, settings_yaml_path: str) -> None:
        self.settings = Settings.from_yaml(settings_yaml_path)
```

# core/audit.py
from __future__ import annotations
import json
from dataclasses import dataclass, asdict
from collections import deque
from pathlib import Path

@dataclass
class AuditEntry:
    timestamp: str
    source: str          # "web" | "mcp"
    actor: str
    action: str
    target: str
    detail: dict
    success: bool

class AuditLogger:
    def __init__(self, log_file: str, memory_size: int = 200):
        self.log_file = Path(log_file)
        self.log_file.parent.mkdir(parents=True, exist_ok=True)
        self._memory: deque[AuditEntry] = deque(maxlen=memory_size)

    def log(self, entry: AuditEntry) -> None:
        self._memory.append(entry)
        with self.log_file.open("a", encoding="utf-8") as f:
            f.write(json.dumps(asdict(entry), ensure_ascii=False) + "\n")

    def recent(self, n: int | None = None) -> list[AuditEntry]:
        items = list(self._memory)
        return items[-n:] if n else items

    def query(self, source: str | None = None, actor: str | None = None,
              limit: int = 100) -> list[AuditEntry]:
        items = list(self._memory)
        if source:
            items = [e for e in items if e.source == source]
        if actor:
            items = [e for e in items if e.actor == actor]
        return items[-limit:]
```

# core/admin_auth.py
from __future__ import annotations
import yaml
import time
from pathlib import Path
from itsdangerous import URLSafeTimedSerializer, BadSignature, SignatureExpired
import bcrypt

class AdminAuth:
    def __init__(self, admin_yaml: str, session_secret: str):
        self.admin_yaml = Path(admin_yaml)
        self.session_secret = session_secret
        self._data = {"username": "admin", "password_hash": ""}
        if self.admin_yaml.exists():
            with self.admin_yaml.open(encoding="utf-8") as f:
                self._data = yaml.safe_load(f) or self._data

    def set_password(self, username: str, password: str) -> None:
        self._data["username"] = username
        self._data["password_hash"] = bcrypt.hashpw(
            password.encode("utf-8"), bcrypt.gensalt()
        ).decode("utf-8")
        self.admin_yaml.parent.mkdir(parents=True, exist_ok=True)
        with self.admin_yaml.open("w", encoding="utf-8") as f:
            yaml.safe_dump(self._data, f, allow_unicode=True)
        try:
            import os; os.chmod(self.admin_yaml, 0o600)
        except OSError:
            pass

    def verify(self, username: str, password: str) -> bool:
        if username != self._data.get("username"):
            return False
        h = self._data.get("password_hash", "")
        if not h:
            return False
        return bcrypt.checkpw(password.encode("utf-8"), h.encode("utf-8"))

class SessionManager:
    def __init__(self, secret: str, timeout: int = 28800):
        self.serializer = URLSafeTimedSerializer(secret, salt="web-session")
        self.timeout = timeout

    def create_token(self, username: str) -> str:
        return self.serializer.dumps({"u": username})

    def validate_token(self, token: str) -> str | None:
        try:
            data = self.serializer.loads(token, max_age=self.timeout)
            return data.get("u")
        except (BadSignature, SignatureExpired):
            return None
```

# core/csrf.py
from __future__ import annotations
import secrets
from itsdangerous import URLSafeTimedSerializer, BadSignature

class CSRFProtection:
    def __init__(self, secret: str):
        self.serializer = URLSafeTimedSerializer(secret, salt="csrf")

    def generate_token(self) -> str:
        return self.serializer.dumps({"t": secrets.token_hex(16)})

    def validate_token(self, token: str) -> bool:
        try:
            self.serializer.loads(token, max_age=86400)
            return True
        except BadSignature:
            return False
```

# web/routes.py
from __future__ import annotations
from pathlib import Path
from starlette.requests import Request
from starlette.responses import Response, JSONResponse, RedirectResponse
from jinja2 import Environment, FileSystemLoader
from core.admin_auth import AdminAuth, SessionManager
from core.csrf import CSRFProtection

TEMPLATE_DIR = Path(__file__).parent / "templates"
STATIC_DIR = Path(__file__).parent / "static"

def register_web_routes(mcp, app_state, admin_auth: AdminAuth,
                        session_mgr: SessionManager, csrf: CSRFProtection):
    env = Environment(loader=FileSystemLoader(str(TEMPLATE_DIR)), autoescape=True)

    def require_login(request: Request) -> str | None:
        token = request.cookies.get("session")
        if not token:
            return None
        return session_mgr.validate_token(token)

    @mcp.custom_route("/ui/login", methods=["GET"])
    async def login_page(request):
        token = csrf.generate_token()
        html = env.get_template("login.html").render(csrf_token=token)
        response = Response(html, media_type="text/html")
        response.set_cookie("csrf_token", token, httponly=True, samesite="strict")
        return response

    @mcp.custom_route("/ui/login", methods=["POST"])
    async def login_submit(request):
        form = await request.form()
        username = form.get("username", "")
        password = form.get("password", "")
        csrf_token = form.get("csrf_token", "")
        if not csrf.validate_token(csrf_token):
            return Response("CSRF 校验失败", status_code=403)
        if admin_auth.verify(username, password):
            token = session_mgr.create_token(username)
            response = RedirectResponse("/ui/", status_code=303)
            response.set_cookie("session", token, httponly=True, samesite="strict")
            return response
        return Response("用户名或密码错误", status_code=401)

    @mcp.custom_route("/ui/logout", methods=["GET"])
    async def logout(request):
        response = RedirectResponse("/ui/login", status_code=303)
        response.delete_cookie("session")
        return response

    @mcp.custom_route("/ui/", methods=["GET"])
    async def dashboard(request):
        user = require_login(request)
        if not user:
            return RedirectResponse("/ui/login", status_code=303)
        html = env.get_template("dashboard.html").render(user=user)
        return Response(html, media_type="text/html")

    # 静态文件
    @mcp.custom_route("/ui/static/{filename}", methods=["GET"])
    async def static_file(request):
        filename = request.path_params["filename"]
        f = STATIC_DIR / filename
        if not f.exists():
            return Response("Not Found", status_code=404)
        return Response(f.read_bytes(), media_type="text/css")
```

# tools/user_mgmt_tools.py
from __future__ import annotations
import yaml, secrets
from pathlib import Path

def _load_auth_yaml(path: str) -> dict:
    with Path(path).open(encoding="utf-8") as f:
        return yaml.safe_load(f) or {}

def _save_auth_yaml(path: str, data: dict) -> None:
    p = Path(path)
    with p.open("w", encoding="utf-8") as f:
        yaml.safe_dump(data, f, allow_unicode=True)
    try:
        import os; os.chmod(p, 0o600)
    except OSError: pass

def create_api_key(auth_yaml: str, name: str, role: str,
                   scope_include: list[str] | None = None,
                   app_state=None) -> dict:
    data = _load_auth_yaml(auth_yaml)
    key = "sk-dm-" + secrets.token_hex(16)
    entry = {"key": key, "name": name, "role": role}
    if scope_include and scope_include != [""]:
        entry["scope"] = {"containers": {"include": [s.strip() for s in scope_include if s.strip()]}}
    data.setdefault("keys", []).append(entry)
    _save_auth_yaml(auth_yaml, data)
    if app_state: app_state.reload_auth(auth_yaml)
    return {"success": True, "key": key, "name": name}

def delete_api_key(auth_yaml: str, name: str, app_state=None) -> dict:
    data = _load_auth_yaml(auth_yaml)
    before = len(data.get("keys", []))
    data["keys"] = [k for k in data.get("keys", []) if k.get("name") != name]
    after = len(data["keys"])
    if before == after:
        return {"success": False, "error": f"未找到名为 {name} 的 Key"}
    _save_auth_yaml(auth_yaml, data)
    if app_state: app_state.reload_auth(auth_yaml)
    return {"success": True, "removed": name}
```

修改 `web/routes.py` 的 `static_file` 函数，按扩展名设置正确的 Content-Type：

```python
    @mcp.custom_route("/ui/static/{filename}", methods=["GET"])
    async def static_file(request):
        filename = request.path_params["filename"]
        f = STATIC_DIR / filename
        if not f.exists():
            return Response("Not Found", status_code=404)
        ext = f.suffix.lower()
        media = {".css": "text/css", ".js": "application/javascript"}.get(ext, "text/plain")
        return Response(f.read_bytes(), media_type=media)
```
